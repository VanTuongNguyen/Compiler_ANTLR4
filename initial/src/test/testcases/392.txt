
        procedure Main();
        var
        toPrint: Integer;
        i: Integer;
        begin
        for i := 0 to 255 do
        begin
            toPrint := Char(i);
            Write(i, " ");
            
            if IsPrintable(toPrint) then
            begin
            Write(toPrint);
            end
            else
            begin
            Write(" ");
            end
            
            if ((i + 1) mod 10) = 0 then
            begin
            WriteLn();
            end
        end
        end
