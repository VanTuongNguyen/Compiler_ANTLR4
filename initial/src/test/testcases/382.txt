
        procedure llRotate(x : integer);
        var y : integer;
        begin
            if x + rr=NULL then exit();
            y := x + rr;
            rr := y + ll;
            if y+ ll <> NULL then y := x;
            y := x+ pp;
            if x<>root then
                if x=x + pp +  ll
                then x := y;
                else x := y;
            else root := y;
            y := x;
            x := y;
        end
