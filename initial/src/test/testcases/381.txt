
        procedure main();
        var a : array [0 .. 1230997] of integer;
            n, key : integer;

            i : integer;
        begin
            readln(n);
            for i := 1 to n do
            read(a[i]);

            while not seekeof do
            begin
                readln(key);
                i := bsearch(a, n, key);
                if i=0 then writeln("Not found");
                else writeln(a[i]);
            end
        end
